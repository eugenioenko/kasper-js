import { Card, CardGrid } from '@astrojs/starlight/components';

Kasper exists because building reactive UIs shouldn't require understanding a compiler, a scheduler, a virtual DOM reconciler, and a hook dependency system.

<CardGrid>
  <Card title="No Build Step Required" icon="rocket">
    Drop a 16KB script tag and start building. Signals, router, slots, and lazy loading all work without a bundler. A build pipeline adds only the `.kasper` single-file component format.
  </Card>
  <Card title="HTML-first Templates" icon="document">
    Directives are standard HTML attributes: `@if`, `@each`, `@on:click`. Write `@if` where you'd write `if`. Templates are readable by anyone who knows HTML.
  </Card>
  <Card title="Automatic Cleanup" icon="approve-check">
    Components are classes. `this.watch()`, `this.effect()`, and `this.computed()` all release automatically on destroy. No dependency arrays, no manual unsubscribe.
  </Card>
  <Card title="Fine-grained Signals" icon="list-format">
    Reactive primitives that update only the exact DOM nodes that depend on them. No virtual DOM, no diffing, no full re-renders.
  </Card>
  <Card title="AI Agent Ready" icon="star">
    Entire API fits in a single context window. Paste [llms.txt](https://kasperjs.top/llms.txt) into any agent and it can write idiomatic Kasper code from scratch.
  </Card>
  <Card title="Rich Expression Language" icon="puzzle">
    Arrow functions, pipeline operator `|>`, optional chaining, spread, and more — a proper AST-based expression interpreter, not eval.
  </Card>
</CardGrid>
